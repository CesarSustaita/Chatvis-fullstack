void calculos()
{
    int num;
    int suma = 0, i,arr[DATMAX],aux,k,pos,posmayor,n,mayor;
    num = pidaDatos(arr);
    int ar[num];
    float prom,me;

//Promedio
    for(i = 0; i<num; i++)
    {
        suma += arr[i];
    }
    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(int j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

void calculos()
{
    int num;
    int suma = 0, i,arr[DATMAX],aux,k,pos,posmayor,n,mayor;
    num = pidaDatos(arr);
    int ar[num];
    float prom,me;

//Promedio
    for(i = 0; i<num; i++)
    {
        suma += arr[i];
    }
    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(int j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

void calculos()
{
    int num;
    int suma = 0,i,j,arr[DATMAX],aux,k,pos,posmayor,n,mayor;
    num = pidaDatos(arr);
    int ar[num];
    float prom,me;

//Promedio
    for(i = 0; i<num; i++)
    {
        suma += arr[i];
    }
    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

void calculos() {
    int num, op;

    do {
        system("cls");
        PrintBanner(8);

        int suma = 0, i, arr[DATMAX], aux, k, pos, posmayor, n, mayor;
        num = pidaDatos(arr);
        int ar[num];
        float prom, me;

        //Promedio
        for(i = 0; i<num; i++) {
            suma += arr[i];
        }

        prom = (float) suma / num;
        printf("\n\t         %c La Media/Promedio es: %.2f", 254, prom);

        //Reordenar arreglo con InsertionSort
        for(i=0; i<num; i++) {
            aux = arr[i];
            k = i - 1;
            while (k >= 0 && arr[k] > aux) {
                arr[k + 1] = arr[k];
                k = k - 1;
            }
            arr[k + 1] = aux;
        }

        //Mediana
        pos = num/2;
        me = 0;
        if(num %2 == 0) {
            me = (float)(arr[pos-1]+arr[pos])/2;
        } else {
            me = arr[pos];
        }
        printf("\n\t         %c La Mediana es: %.2f", 254, me);

        //Moda
        n = 0;
        posmayor = 0;
        mayor=0;

        for(i = 0; i<num; i++) {
            ar[i]=0;
        }

        for(i = 0; i < num; i++) {
            n = arr[i];
            pos = i;
            for(int j = i; j < num; j++) {
                if(arr[j] == n) {
                    ar[pos]++;
                }
            }
        }

        mayor = ar[0];
        posmayor = 0;

        for(i=0;i<num;i++) {
            if(ar[i]>mayor) {
                posmayor=i;
                mayor=ar[i];
            }
        }

        if(mayor == 1) {
            printf("\n\t         %c No hay moda", 254);
        } else {
            printf("\n\t         %c La Moda es: %d", 254, arr[posmayor]);
        }
        op = Select();
    } while( op == 1);
};

    } else {

void calculos() {
    int num, op;

    do {
        system("cls");
        PrintBanner(8);

        int suma = 0, i, arr[DATMAX], aux, k, pos, posmayor, n, mayor;
        num = pidaDatos(arr);
        int ar[num];
        float prom, me;

        //Promedio
        for(i = 0; i<num; i++) {
            suma += arr[i];
        }

        prom = (float) suma / num;
        printf("\n\t         %c La Media/Promedio es: %.2f", 254, prom);

        //Reordenar arreglo con InsertionSort
        for(i=0; i<num; i++) {
            aux = arr[i];
            k = i - 1;
            while (k >= 0 && arr[k] > aux) {
                arr[k + 1] = arr[k];
                k = k - 1;
            }
            arr[k + 1] = aux;
        }

        //Mediana
        pos = num/2;
        me = 0;
        if(num %2 == 0) {
            me = (float)(arr[pos-1]+arr[pos])/2;
        } else {
            me = arr[pos];
        }
        printf("\n\t         %c La Mediana es: %.2f", 254, me);

        //Moda
        n = 0;
        posmayor = 0;
        mayor=0;

        for(i = 0; i<num; i++) {
            ar[i]=0;
        }

        for(i = 0; i < num; i++) {
            n = arr[i];
            pos = i;
            for(int j = i; j < num; j++) {
                if(arr[j] == n) {
                    ar[pos]++;
                }
            }
        }

        mayor = ar[0];
        posmayor = 0;

        for(i=0;i<num;i++) {
            if(ar[i]>mayor) {
                posmayor=i;
                mayor=ar[i];
            }
        }

        if(mayor == 1) {
            printf("\n\t         %c No hay moda", 254);
        } else {
            printf("\n\t         %c La Moda es: %d", 254, arr[posmayor]);
        }
        op = Select();
    } while( op == 1);
};

int factorialR(int x) {
    if (x==0) {
        return 1;
    } else {
        return x*factorialR(x-1);
    }
};
