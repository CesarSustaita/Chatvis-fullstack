/* 	Se puede llamar directamente a la función AritmeticaP si
	se tienen los parámetros para realizar las operaciones, de
	lo contrario se puede llamar a la funcion Aritmetica y esta 
	se encargará de solicitar los datos al usuario.
	
	Los parámetros son los siguientes.

	int operacion
		1 -> suma
		2 -> resta
		3 -> multiplicación
		4 -> división
	float operando1
	float operando2

	La resta y división se calculan de la manera siguiente:
	operando1 - operando2;
	operando1 / operando2;

	La funcion AritmeticaP regresa el valor del resultado.
*/

float AritmeticaP(int operacion, float operando1, float operando2){
	switch(operacion){
		case 1:
			return(operando1+operando2);
			break;
		case 2:
			return(operando1-operando2);
			break;
		case 3:
			return(operando1*operando2);
			break;
		case 4:
			if(operando2 != 0){
				return(operando1/operando2);
			} else {
				printf("La division por 0 es indeterminada.\n");
				return 0;
			}
			break;
		default:
			printf("Operacion no especificada.\n");
			return 0;			
			break;
	}
}

void calculos()
{
    int num;
    int suma = 0, i,arr[DATMAX],aux,k,pos,posmayor,n,mayor;
    num = pidaDatos(arr);
    int ar[num];
    float prom,me;

//Promedio
    for(i = 0; i<num; i++)
    {
        suma += arr[i];
    }
    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(int j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

void calculos()
{

    {
        suma += arr[i];

    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(int j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;

        me = arr[pos];

    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(int j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");

        printf("\nLa Moda es: %d", arr[posmayor]);

void calculos()
{

    int suma = 0,i,j,arr[DATMAX],aux,k,pos,posmayor,n,mayor;
    num = pidaDatos(arr);
    int ar[num];
    float prom,me;

    {
        suma += arr[i];
    }
    prom = (float) suma / num;
    printf("\n\nLa Media/Promedio es: %.2f",prom);

//Reordenar arreglo con InsertionSort
    for(i=0; i<num; i++)
    {
        aux = arr[i];
        k = i - 1;
        while (k >= 0 && arr[k] > aux)
        {
            arr[k + 1] = arr[k];
            k = k - 1;
        }
        arr[k + 1] = aux;
    }

//Mediana
    pos = num/2;
    me = 0;
    if(num %2 == 0)
        me = (float)(arr[pos-1]+arr[pos])/2;
    else
        me = arr[pos];

    printf("\nLa Mediana es: %.2f", me);

//Moda
    n = 0;
    posmayor = 0;
    mayor=0;

    for(i = 0; i<num; i++)
    {
        ar[i]=0;
    }

    for(i=0;i<num;i++)
    {
        n = arr[i];
        pos = i;
        for(j = i; j < num; j++)
        {
            if(arr[j] == n)
                ar[pos]++;
        }
    }
    mayor=ar[0];
    posmayor = 0;
    for(i=0;i<num;i++) {
        if(ar[i]>mayor) {
            posmayor=i;
            mayor=ar[i];
        }
    }
    if(mayor == 1)
        printf("\nNo hay moda");
    else
        printf("\nLa Moda es: %d", arr[posmayor]);

}

void calculos() {

        num = pidaDatos(arr);
        int ar[num];
        float prom, me;

        //Promedio
        for(i = 0; i<num; i++) {
            suma += arr[i];
        }

        prom = (float) suma / num;
        printf("\n\t         %c La Media/Promedio es: %.2f", 254, prom);

        //Reordenar arreglo con InsertionSort
        for(i=0; i<num; i++) {
            aux = arr[i];
            k = i - 1;
            while (k >= 0 && arr[k] > aux) {
                arr[k + 1] = arr[k];
                k = k - 1;
            }
            arr[k + 1] = aux;
        }

        //Mediana
        pos = num/2;
        me = 0;
        if(num %2 == 0) {
            me = (float)(arr[pos-1]+arr[pos])/2;
        } else {
            me = arr[pos];
        }
        printf("\n\t         %c La Mediana es: %.2f", 254, me);

        //Moda
        n = 0;
        posmayor = 0;
        mayor=0;

        for(i = 0; i<num; i++) {
            ar[i]=0;
        }

        for(i = 0; i < num; i++) {
            n = arr[i];
            pos = i;
            for(int j = i; j < num; j++) {
                if(arr[j] == n) {
                    ar[pos]++;
                }
            }
        }

        mayor = ar[0];
        posmayor = 0;

        for(i=0;i<num;i++) {
            if(ar[i]>mayor) {
                posmayor=i;
                mayor=ar[i];
            }
        }

        if(mayor == 1) {
            printf("\n\t         %c No hay moda", 254);
        } else {
            printf("\n\t         %c La Moda es: %d", 254, arr[posmayor]);
        }
        op = Select();
    } while( op == 1);
};

void calculos() {
    int num, op;

    do {
        system("cls");
        PrintBanner(8);

        int suma = 0, i, arr[DATMAX], aux, k, pos, posmayor, n, mayor;
        num = pidaDatos(arr);
        int ar[num];
        float prom, me;

        //Promedio
        for(i = 0; i<num; i++) {
            suma += arr[i];
        }

        prom = (float) suma / num;
        printf("\n\t         %c La Media/Promedio es: %.2f", 254, prom);

        //Reordenar arreglo con InsertionSort
        for(i=0; i<num; i++) {
            aux = arr[i];
            k = i - 1;
            while (k >= 0 && arr[k] > aux) {
                arr[k + 1] = arr[k];
                k = k - 1;
            }
            arr[k + 1] = aux;
        }

        //Mediana
        pos = num/2;
        me = 0;
        if(num %2 == 0) {
            me = (float)(arr[pos-1]+arr[pos])/2;
        } else {
            me = arr[pos];
        }
        printf("\n\t         %c La Mediana es: %.2f", 254, me);

        //Moda
        n = 0;
        posmayor = 0;
        mayor=0;

        for(i = 0; i<num; i++) {
            ar[i]=0;
        }

        for(i = 0; i < num; i++) {
            n = arr[i];
            pos = i;
            for(int j = i; j < num; j++) {
                if(arr[j] == n) {
                    ar[pos]++;
                }
            }
        }

        mayor = ar[0];
        posmayor = 0;

        for(i=0;i<num;i++) {
            if(ar[i]>mayor) {
                posmayor=i;
                mayor=ar[i];
            }
        }

        if(mayor == 1) {
            printf("\n\t         %c No hay moda", 254);
        } else {
            printf("\n\t         %c La Moda es: %d", 254, arr[posmayor]);
        }
        op = Select();
    } while( op == 1);
};

    if (x==0) {
        return 1;
    } else {

};
